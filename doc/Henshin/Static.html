def write
  write_path = File.join( config[:root], config[:target], @path[ config[:root].size..-1 ] )
  
  FileUtils.mkdir_p File.join( write_path.directory )
  file = File.new( File.join( write_path ), "w" )
  file.puts( @content )
end
